                ]),
                'location' => ShelfResource::make($this->shelves),
                $this->mergeWhen($this->whenLoaded('shelves'), [
                    'location' => ShelfResource::make($this->shelves),
                ]),

            ];
        }

namespace App\Http\Resources;

use Illuminate\Http\Resources\Json\JsonResource;

class ShelfResource extends JsonResource
{
    public function toArray($request)
    {
        $bookshelf = $this->bookshelves;
        $room = $bookshelf?->rooms;
        $building = $room?->buildings;

        return [
            $this->node($building, 'building', 'pi pi-fw pi-home', [
                $this->node($room, 'room', 'pi pi-fw pi-stop', [
                    $this->node($bookshelf, 'bookshelf', 'pi pi-fw pi-server', [
                        $this->node($this, 'shelf', 'pi pi-fw pi-book'),
                    ]),
                ]),
            ]),
        ];
    }

    private function node($model, string $type, string $icon, array $children = []): array
    {
        return [
            'id' => $model?->id,
            'key' => $type . '-' . $model?->id,
            'label' => $model?->name,
            'data' => $model?->name,
            'icon' => $icon,
            'children' => $children,
        ];
    }
