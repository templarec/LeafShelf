===== app/Http/Controllers/BookController.php =====
<?php

namespace App\Http\Controllers;

use App\Http\Resources\BookCollection;
use App\Http\Resources\ShelfCollection;
use App\Http\Resources\AuthorCollection;
use App\Http\Resources\ShelfResource;
use App\Models\Author;
use App\Models\Book;
use App\Models\Building;
use App\Models\Shelf;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\DB;

class BookController extends Controller
{
    public function getLocation(Request $request)
    {
        return new ShelfResource(
            Shelf::with('bookshelves.rooms.buildings')->find($request->idShelf)
        );
    }

    public function saveBook(Request $request)
    {
        DB::transaction(function () use ($request) {
            $book = Book::create([
                'ISBN' => $request->isbn,
                'title' => $request->title,
                'publisher' => $request->publisher,
                'pages' => $request->pages,
                'img' => $request->img,
                'shelf_id' => $request->shelfId,
            ]);

            foreach ($request->authors as $autori) {
                $author = Author::firstOrCreate([
                    'name' => trim($autori['nome']),
                    'surname' => trim($autori['cognome']),
                ]);

                $book->authors()->syncWithoutDetaching($author->id);
            }
        });
    }
    public function all()
    {
        return new BookCollection(Book::all());

    }

    public function search(Request $request)
    {
        $searchterm = $request->searchtxt;

        return new BookCollection(Book::where('title', 'like', "%{$searchterm}%")->get());
    }

    public function lastBooks()
    {
        return new BookCollection(Book::orderBy('id', 'desc')->take(5)->get());
    }

}

        $key = env('ISBNdb_KEY');
        $header = [
            'Accept' => '*/*',
            'Authorization' => $key,
        ];

        $response = Http::withHeaders($header)->get($url);
        return [
            $response->json()];
    }
}

===== app/Http/Controllers/ShelfController.php =====
<?php

namespace App\Http\Controllers;

use App\Http\Resources\BookshelfResource;
use App\Http\Resources\EdificioCollection;
use App\Http\Resources\EdificioResource;
use App\Http\Resources\RipianoResource;
use App\Http\Resources\RoomResource;
use App\Http\Resources\ScaffaleResource;
use App\Http\Resources\ShelfResource;
use App\Http\Resources\StanzaResource;
use App\Models\Bookshelf;
use App\Models\Building;
use App\Models\Room;
use App\Models\Shelf;
use Illuminate\Http\Request;
use ProtoneMedia\LaravelCrossEloquentSearch\Search;

class ShelfController extends Controller
{
    public function getShelves(Request $request)
    {
        $response = Shelf::getShelves($request->idBookshelf);

        return response()->json($response);
    }

    public function addShelf(Request $request)
    {
        $shelf = Shelf::create([
            'name' => $request->name,
            'bookshelf_id' => $request->bookshelf_id,
        ]);

        $shelf->update([
            'key' => 'shelf-' . $shelf->id
        ]);
    }

    public function getPosKey(Request $request)
    {
        if (!str_starts_with($request->keyRip, 'shelf-')) {
            return response()->json([
                'success' => false,
                'message' => 'Chiave ripiano non valida',
            ], 400);
        }

        $id = (int) str_replace('shelf-', '', $request->keyRip);
        $shelf = Shelf::find($id);

        if (!$shelf) {
            return response()->json([
                'success' => false,
                'message' => 'Ripiano non trovato',
            ], 404);
        }

        return ShelfResource::make($shelf);
    }

    public function getPosKeyGlobal(Request $request)
    {
        $key = $request->keyRip;

        $map = [
            'room-' => [Room::class, StanzaResource::class, true, 'Room not found'],
            'bookshelf-' => [Bookshelf::class, ScaffaleResource::class, true, 'Bookshelf not found'],
            'shelf-' => [Shelf::class, ShelfResource::class, false, 'Shelf not found'],
        ];

        foreach ($map as $prefix => [$modelClass, $resourceClass, $selectable, $notFoundMessage]) {
            if (str_starts_with($key, $prefix)) {
                $id = (int) substr($key, strlen($prefix));
                $model = $modelClass::find($id);

                if (!$model) {
                    return response()->json(['message' => $notFoundMessage], 404);
                }

                $resource = $resourceClass::make($model);

                if ($selectable) {
                    $resource->selectable(true);
                }

                return $resource;
            }
        }

        return response()->json([
            'message' => 'Chiave non valida'
        ], 400);
    }

    public function delItem(Request $request)
    {
        try {
            $allowedModels = [
                Building::class,
                Room::class,
                Bookshelf::class,
                Shelf::class,
            ];

            $model = $request->model;

            if (!in_array($model, $allowedModels, true)) {
                return response()->json([
                    'success' => false,
                    'message' => 'Modello non consentito',
                ], 400);
            }

            $delitem = $model::destroy($request->id);

            return response()->json([
                'success' => true,
                'message' => 'Cancellato con successo',
                'output' => $delitem,
            ], 200);
        } catch (\Exception $e) {
            return response()->json([
                'success' => false,
                'message' => $e->getMessage(),
            ], 500);
        }
    }
    public function getTree(Request $request)
{
    $selectable = $request->boolean('selectable');

    return EdificioCollection::make(
        Building::query()
            ->with('rooms.bookshelves.shelves')
            ->get()
    )->selectable($selectable);
}
}

class Author extends Model
{
    use HasFactory;

    protected $fillable = [
        'name',
        'surname',
    ];

    public function books()
    {
        return $this->belongsToMany(Book::class, 'books_authors');
    }

    public static function findAuthor($name, $surname)
    {
        return self::where('name', trim($name))
            ->where('surname', trim($surname))
            ->first();
    }
